3.membuat fungsi Async/await nya. jika memakai promise harus pakai metode then

async function contohAsync () {
    let p = new Promise((resolve, reject) => {
        setTimeout(() => resolve("Hello World!"), 2000)
    });

    let result = await p;
    alert(result);
}

contohAsync();
